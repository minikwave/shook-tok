1. 제품 범위
SHOOK은 서비스/브랜드명이다.
TOK은 MVP에서 구현할 핵심 액션이다.

MVP 목표는 단순하다.

디스코드 서버 안에서 유저가 다른 유저에게 가볍게 TOK을 보내고, 상대가 응답하면 포인트와 스트릭이 쌓이는 봇.

이번 MVP에서는 광고, AI, 유료 결제, 스트리머 오버레이, 익명 TOK, 캠페인 기능은 구현하지 않는다.

CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  discord_user_id TEXT NOT NULL UNIQUE,
  username TEXT,
  avatar_url TEXT,
  points_balance INTEGER NOT NULL DEFAULT 0,
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMP NOT NULL DEFAULT NOW()
);
servers
CREATE TABLE servers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  discord_guild_id TEXT NOT NULL UNIQUE,
  name TEXT,
  default_channel_id TEXT,
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMP NOT NULL DEFAULT NOW()
);
pokes
내부 테이블명은 pokes로 두되, UI에서는 항상 TOK이라고 부른다.

CREATE TABLE pokes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  server_id UUID NOT NULL REFERENCES servers(id),
  sender_id UUID NOT NULL REFERENCES users(id),
  receiver_id UUID NOT NULL REFERENCES users(id),
  emotion TEXT,
  message TEXT,
  status TEXT NOT NULL DEFAULT 'pending',
  created_at TIMESTAMP NOT NULL DEFAULT NOW(),
  responded_at TIMESTAMP
);
status:

pending
responded
expired
cancelled
point_ledger
CREATE TABLE point_ledger (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id),
  server_id UUID NOT NULL REFERENCES servers(id),
  amount INTEGER NOT NULL,
  reason TEXT NOT NULL,
  ref_type TEXT,
  ref_id UUID,
  created_at TIMESTAMP NOT NULL DEFAULT NOW()
);

16. 사람이 참고할 제품 컨텍스트
MVP 이후 SHOOK은 다음 방향으로 확장 가능하다.

SHOOK = 전체 서비스/플랫폼
TOK = 기본 관심 신호
Super TOK = 유료 강조 신호
AI TOK = 관계 유지 추천
Sponsored TOK = 광고/브랜드 캠페인
Stream TOK = 스트리머/팬덤 기능
하지만 현재 개발자는 이 확장 기능을 구현하지 않는다.
현재는 TOK MVP의 사용감, 반복성, 재미, 서버 내 랭킹만 검증한다.

import { User as DiscordUser } from "discord.js";
import { prisma } from "@shook/db";
import { TOK_LIMITS, TOK_POINTS } from "@shook/shared";

type SendTokInput = {
  guildId: string;
  guildName: string;
  sender: DiscordUser;
  receiver: DiscordUser;
  emotion?: string;
  message?: string | null;
};

export async function sendTok(input: SendTokInput) {
  const { guildId, guildName, sender, receiver, emotion, message } = input;

  if (sender.id === receiver.id) {
    return {
      ok: false,
      message: "자기 자신에게는 TOK할 수 없어요.",
    };
  }

  if (receiver.bot) {
    return {
      ok: false,
      message: "봇에게는 TOK할 수 없어요.",
    };
  }

  const server = await prisma.server.upsert({
    where: { discordGuildId: guildId },
    update: { name: guildName },
    create: { discordGuildId: guildId, name: guildName },
  });

  const senderUser = await upsertDiscordUser(sender);
  const receiverUser = await upsertDiscordUser(receiver);

  const cooldownSince = new Date(
    Date.now() - TOK_LIMITS.SAME_RECEIVER_COOLDOWN_MINUTES * 60 * 1000
  );

  const recentSameReceiverTok = await prisma.poke.findFirst({
    where: {
      serverId: server.id,
      senderId: senderUser.id,
      receiverId: receiverUser.id,
      createdAt: { gte: cooldownSince },
    },
  });

  if (recentSameReceiverTok) {
    return {
      ok: false,
      message: "아직 같은 친구에게 다시 TOK할 수 없어요. 조금 뒤에 다시 시도해주세요.",
    };
  }

  const dailySince = new Date(Date.now() - 24 * 60 * 60 * 1000);

  const dailyTokCount = await prisma.poke.count({
    where: {
      serverId: server.id,
      senderId: senderUser.id,
      createdAt: { gte: dailySince },
    },
  });

  if (dailyTokCount >= TOK_LIMITS.DAILY_INITIATED_TOK_LIMIT) {
    return {
      ok: false,
      message: "오늘 먼저 보낼 수 있는 TOK을 모두 사용했어요.",
    };
  }

  const poke = await prisma.poke.create({
    data: {
      serverId: server.id,
      senderId: senderUser.id,
      receiverId: receiverUser.id,
      emotion,
      message,
      status: "PENDING",
    },
  });

  await grantPoints({
    userId: senderUser.id,
    serverId: server.id,
    amount: TOK_POINTS.SENT,
    reason: "tok_sent",
    refType: "poke",
    refId: poke.id,
  });

  await grantPoints({
    userId: receiverUser.id,
    serverId: server.id,
    amount: TOK_POINTS.RECEIVED,
    reason: "tok_received",
    refType: "poke",
    refId: poke.id,
  });

  return {
    ok: true,
    data: {
      pokeId: poke.id,
      senderName: sender.username,
      receiverName: receiver.username,
      emotion: emotion ?? "🔥",
      message,
      remainingToday: TOK_LIMITS.DAILY_INITIATED_TOK_LIMIT - dailyTokCount - 1,
    },
  };
}

type TokBackInput = {
  guildId: string;
  responder: DiscordUser;
  sender: DiscordUser;
};

export async function tokBack(input: TokBackInput) {
  const { guildId, responder, sender } = input;

  const server = await prisma.server.findUnique({
    where: { discordGuildId: guildId },
  });

  if (!server) {
    return {
      ok: false,
      message: "아직 이 서버에서 TOK 기록이 없어요.",
    };
  }

  const responderUser = await upsertDiscordUser(responder);
  const senderUser = await upsertDiscordUser(sender);

  const expireSince = new Date(
    Date.now() - TOK_LIMITS.TOK_EXPIRE_MINUTES * 60 * 1000
  );

  const pendingTok = await prisma.poke.findFirst({
    where: {
      serverId: server.id,
      senderId: senderUser.id,
      receiverId: responderUser.id,
      status: "PENDING",
      createdAt: { gte: expireSince },
    },
    orderBy: { createdAt: "desc" },
  });

  if (!pendingTok) {
    return {
      ok: false,
      message: "응답할 수 있는 TOK을 찾지 못했어요.",
    };
  }

  const now = new Date();
  const responseSeconds = Math.floor(
    (now.getTime() - pendingTok.createdAt.getTime()) / 1000
  );

  await prisma.poke.update({
    where: { id: pendingTok.id },
    data: {
      status: "RESPONDED",
      respondedAt: now,
    },
  });

  const withinBonusWindow =
    responseSeconds <= TOK_LIMITS.RESPONSE_BONUS_WINDOW_MINUTES * 60;

  if (withinBonusWindow) {
    await grantPoints({
      userId: responderUser.id,
      serverId: server.id,
      amount: TOK_POINTS.TOKBACK_BONUS,
      reason: "tokback_bonus",
      refType: "poke",
      refId: pendingTok.id,
    });

    await grantPoints({
      userId: senderUser.id,
      serverId: server.id,
      amount: TOK_POINTS.TOKBACK_BONUS,
      reason: "tokback_bonus",
      refType: "poke",
      refId: pendingTok.id,
    });
  }

  const relationship = await updateRelationship({
    serverId: server.id,
    userAId: senderUser.id,
    userBId: responderUser.id,
  });

  return {
    ok: true,
    data: {
      senderName: sender.username,
      responderName: responder.username,
      responseSeconds,
      bonusGranted: withinBonusWindow,
      bonusPoints: withinBonusWindow ? TOK_POINTS.TOKBACK_BONUS : 0,
      streakCount: relationship.streakCount,
    },
  };
}

async function upsertDiscordUser(user: DiscordUser) {
  return prisma.user.upsert({
    where: { discordUserId: user.id },
    update: {
      username: user.username,
      avatarUrl: user.displayAvatarURL(),
    },
    create: {
      discordUserId: user.id,
      username: user.username,
      avatarUrl: user.displayAvatarURL(),
    },
  });
}

async function grantPoints({
  userId,
  serverId,
  amount,
  reason,
  refType,
  refId,
}: {
  userId: string;
  serverId: string;
  amount: number;
  reason: string;
  refType?: string;
  refId?: string;
}) {
  await prisma.$transaction([
    prisma.pointLedger.create({
      data: {
        userId,
        serverId,
        amount,
        reason,
        refType,
        refId,
      },
    }),
    prisma.user.update({
      where: { id: userId },
      data: {
        pointsBalance: {
          increment: amount,
        },
      },
    }),
  ]);
}

MVP에서는 아주 단순하게 갑니다.

async function updateRelationship({
  serverId,
  userAId,
  userBId,
}: {
  serverId: string;
  userAId: string;
  userBId: string;
}) {
  const [a, b] = [userAId, userBId].sort();

  const existing = await prisma.relationship.findUnique({
    where: {
      serverId_userAId_userBId: {
        serverId,
        userAId: a,
        userBId: b,
      },
    },
  });

  const now = new Date();

  if (!existing) {
    return prisma.relationship.create({
      data: {
        serverId,
        userAId: a,
        userBId: b,
        streakCount: 1,
        totalInteractions: 1,
        lastInteractionAt: now,
        lastStreakDate: now,
      },
    });
  }

  const last = existing.lastStreakDate;
  let nextStreak = existing.streakCount;

  if (last) {
    const diffHours = (now.getTime() - last.getTime()) / (1000 * 60 * 60);

    if (diffHours >= 24 && diffHours <= 48) {
      nextStreak += 1;
    } else if (diffHours > 48) {
      nextStreak = 1;
    }
  }

  return prisma.relationship.update({
    where: { id: existing.id },
    data: {
      streakCount: nextStreak,
      totalInteractions: { increment: 1 },
      lastInteractionAt: now,
      lastStreakDate: now,
    },
  });
}
